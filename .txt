    function populateMonthYearSelectors() {
        const yearSelect = $('#meterYearSelect'); // ใช้ jQuery
        const currentYear = new Date().getFullYear();
        const currentThaiYear = currentYear + 543;

        yearSelect.empty();
        for (let i = 0; i <= 2; i++) {
            const year = currentThaiYear - i;
            yearSelect.append(new Option(year, year));
        }
        yearSelect.val(currentThaiYear);

        // เริ่มต้น Select2 สำหรับ Year (มีช่องค้นหา)
        yearSelect.select2({
            theme: "bootstrap-5",
            width: '100%',
            placeholder: "เลือกปี..."
        });

        // เริ่มต้น Select2 สำหรับ Month (มีช่องค้นหา)
        $('#meterMonthSelect').select2({
            theme: "bootstrap-5",
            width: '100%',

        });

        // เรียกฟังก์ชันเพื่อสร้าง Option เดือน
        updateAvailableMonths();
    }

    function updateAvailableMonths(forceIncludeMonth = null) {
        const monthSelect = $('#meterMonthSelect'); // ใช้ jQuery
        const yearSelect = $('#meterYearSelect'); // ใช้ jQuery

        const selectedThaiYear = parseInt(yearSelect.val(), 10);
        const selectedYearAD = selectedThaiYear - 543;

        const thaiMonths = [
            "มกราคม", "กุมภาพันธ์", "มีนาคม", "เมษายน", "พฤษภาคม", "มิถุนายน",
            "กรกฎาคม", "สิงหาคม", "กันยายน", "ตุลาคม", "พฤศจิกายน", "ธันวาคม"
        ];

        // หาเดือนที่มีข้อมูลอยู่แล้วในปีที่เลือก
        const existingMonths = new Set();
        allMeterReadings.forEach(group => {
            if (!group.monthYear) return;
            const [m, y] = group.monthYear.split('/');
            if (parseInt(y, 10) === selectedYearAD) {
                existingMonths.add(parseInt(m, 10));
            }
        });

        // ล้าง Option เก่า
        monthSelect.empty();

        // [สำคัญ] เพิ่ม Option ว่างไว้เป็นอันแรก เพื่อให้ Placeholder ทำงาน
        monthSelect.append('<option></option>');

        let hasOption = false;

        thaiMonths.forEach((monthName, index) => {
            const monthNum = index + 1;
            const monthValue = String(monthNum).padStart(2, '0');

            const isExisting = existingMonths.has(monthNum);
            const isForced = forceIncludeMonth && parseInt(forceIncludeMonth) === monthNum;

            if (!isExisting || isForced) {
                const option = new Option(monthName, monthValue, false, false);
                monthSelect.append(option);
                hasOption = true;
            }
        });

        if (!hasOption && !forceIncludeMonth) {
            // ถ้าเต็มทุกเดือนแล้ว
            monthSelect.prop("disabled", true);
        } else if (!forceIncludeMonth) {
            monthSelect.prop("disabled", false);
        }

        // แจ้ง Select2 ว่ามีการเปลี่ยนแปลง (แต่ไม่ต้อง trigger change เพื่อกัน loop)
        // ใช้แค่การวาดใหม่ก็พอ
    }

    function renderMeterGroups() {
        const container = document.getElementById('meter-group-container');
        container.className = `mt-3 view-${currentMeterView}`;

        const searchTerm = document.getElementById('searchMeterGroup').value.toLowerCase();
        let groups = allMeterReadings.filter(Boolean);

        if (searchTerm) {
            groups = groups.filter(group => {
                const text = monthYearToText(group.monthYear);
                return text.toLowerCase().includes(searchTerm);
            });
        }

        if (groups.length === 0) {
            container.innerHTML = '<div class="text-center text-muted p-5">ไม่พบรายการจดบันทึก</div>';
            return;
        }

        groups.sort((a, b) => b.monthYear.localeCompare(a.monthYear));

        const html = groups.map(group => {
            const readings = group.data || [];

            // นับห้องที่ขาดบันทึก จากข้อมูล Snapshot
            const pendingCount = readings.filter(r => {
                const isWaterActive = r.waterActive !== false;
                const isElectricActive = r.electricActive !== false;
                if (!isWaterActive && !isElectricActive) return false;
                const waterPending = isWaterActive && (r.waterReading === null || r.waterReading === "" || r.waterReading === undefined);
                const electricPending = isElectricActive && (r.electricReading === null || r.electricReading === "" || r.electricReading === undefined);
                return waterPending || electricPending;
            }).length;

            const groupTitle = monthYearToText(group.monthYear);
            const firstRecord = group.createdAt ? new Date(group.createdAt).toLocaleDateString('th-TH') : '-';
            const lastUpdate = group.updatedAt ? new Date(group.updatedAt).toLocaleDateString('th-TH') : '-';

            // จัดรูปแบบตัวเลข
            const pendingDisplay = pendingCount > 0
                ? `<span class="value pending text-danger">${pendingCount} ห้อง</span>`
                : `<span class="value text-success"><i class="fas fa-check-circle"></i> ครบ</span>`;

            return `
     <div class="meter-group-card" data-month-year="${group.monthYear}">
     <div class="meter-group-card-header">
     <h5>${groupTitle}</h5>
     </div>
     <div class="meter-group-card-body">
     <div class="stat-item">
      <span class="label"><i class="fas fa-exclamation-triangle me-2 text-danger"></i>ขาดบันทึก</span>
      ${pendingDisplay}
     </div>
     <div class="stat-item">
      <span class="label"><i class="fas fa-calendar-plus me-2 text-success"></i>เริ่มบันทึก</span>
      <span class="value">${firstRecord}</span>
     </div>
     <div class="stat-item">
      <span class="label"><i class="fas fa-calendar-check me-2 text-primary"></i>ล่าสุด</span>
      <span class="value">${lastUpdate}</span>
     </div>
     </div>
     </div>
     `;
        }).join('');

        container.innerHTML = html;

        container.querySelectorAll('.meter-group-card').forEach(card => {
            card.addEventListener('click', () => {
                showMeterDetailView(card.dataset.monthYear);
            });
        });
    }

function renderMeterDetailTable(monthYear) {
        const container = $('#meter-detail-table-container');
        container.parent().parent().addClass('table-mainstyle');

        const groupData = allMeterReadings.find(g => g.monthYear === monthYear);

        let roomsToRender = [];
        if (groupData && groupData.data && groupData.data.length > 0) {
            roomsToRender = groupData.data.map(reading => {
                const realRoom = window.allRooms.find(r => r.id === reading.roomId);
                return {
                    id: reading.roomId,
                    // ✅ [แก้ไข] ถ้ามีห้องจริงใช้เลขห้องจริง ถ้าไม่มี(ถูกลบ) ใช้เลขห้องที่บันทึกไว้ใน reading
                    number: realRoom ? realRoom.number : (reading.roomNumber || 'ห้องเก่า(ลบแล้ว)'),
                    towerId: reading.towerId || (realRoom ? realRoom.towerId : 'deleted'),
                    tenantNameSnapshot: reading.tenantName,
                    meterWaterActive: reading.waterActive !== undefined ? reading.waterActive : true,
                    meterElectricActive: reading.electricActive !== undefined ? reading.electricActive : true,
                    isDeleted: !realRoom // Flag ไว้บอกว่าห้องนี้ไม่มีในระบบปัจจุบันแล้ว
                };
            });
        } else {
            roomsToRender = window.allRooms;
        }

        if (roomsToRender.length === 0) {
            container.html('<tr><td colspan="100%" class="text-center text-muted p-5">ไม่พบข้อมูลบันทึก</td></tr>');
            return;
        }

        const readings = groupData?.data || [];
        const readingsMap = new Map(readings.map(r => [r.roomId, r]));

        // --- Filter Logic ---
        const searchTerm = document.getElementById('searchMeterDetailRoom').value.toLowerCase();
        const selectedTowers = $('#filterMeterDetailTower').val();
        const filteredRooms = roomsToRender.filter(room => {
            if (selectedTowers && selectedTowers.length > 0) {
                const towerId = room.towerId || 'unassigned';
                if (room.isDeleted && !selectedTowers.includes('deleted') && !selectedTowers.includes(towerId)) {
                   // ถ้าเป็นห้องที่ถูกลบและไม่ได้เลือก filter 'deleted' หรือ tower เดิม ให้ผ่านไปก่อน (หรือจะจัดการแบบอื่นก็ได้)
                } else if (!selectedTowers.includes(towerId)) return false;
            }
            if (searchTerm) {
                const roomNum = String(room.number).toLowerCase();
                const tName = (room.tenantNameSnapshot || '').toLowerCase();
                if (!roomNum.includes(searchTerm) && !tName.includes(searchTerm)) return false;
            }
            return true;
        });

        if (filteredRooms.length === 0) {
            container.html('<tr><td colspan="100%" class="text-center text-muted p-5">ไม่พบห้องพักที่ตรงตามเงื่อนไข</td></tr>');
            return;
        }

        const sortedRooms = filteredRooms.sort((a, b) => String(a.number).localeCompare(String(b.number), undefined, { numeric: true }));

        // ... [ส่วน Helper Functions createMeterHtml, createUsageHtml คงเดิม] ...
        const createMeterHtml = (active, val) => {
            if (!active) return '<span class="meter-disabled-text">ปิดใช้งาน</span>';
            if (val === null || val === undefined || val === '' || Number.isNaN(Number(val))) return '-';
            return parseInt(val, 10);
        };
        const createUsageHtml = (active, currentUsage, trendHtml = '') => {
            if (!active) return '<span class="meter-disabled-text">ปิดใช้งาน</span>';
            if (currentUsage !== null && !isNaN(currentUsage)) {
                return `<div class="d-flex align-items-center justify-content-end"><span class="room-cell fw-bold">${currentUsage}</span>${trendHtml}</div>`;
            }
            return '-';
        };

        // --- Header ---
      const columnConfig = [
            { header: 'เลขห้อง', key: 'roomNumber', class: 'text-center fw-bold', pinnable: true },
            { header: 'ผู้เช่า', key: 'tenant', pinnable: true },
            
            { header: 'ไฟก่อน', key: 'electricPrev', class: 'text-end', pinnable: true },
            { header: 'ไฟนี้', key: 'electricCurr', class: 'text-end', pinnable: true },
            { header: 'หน่วยไฟ', key: 'electricUnit', class: 'text-end', pinnable: true },
            { header: 'น้ำก่อน', key: 'waterPrev', class: 'text-end', pinnable: true },
            { header: 'น้ำนี้', key: 'waterCurr', class: 'text-end', pinnable: true },
            { header: 'หน่วยน้ำ', key: 'waterUnit', class: 'text-end', pinnable: true },
            // ------------------------------------
            { header: 'อัปเดตล่าสุด', key: 'lastUpdated', class: 'text-end', pinnable: true }
        ];

        // ... [ส่วนจัดเรียง Column Header คงเดิม] ...
        const leftPinnedCols = []; const rightPinnedCols = []; const normalCols = [];
        columnConfig.forEach(col => {
            const pinState = meterDetailTablePinningState[col.header];
            if (pinState === 'left') leftPinnedCols.push(col);
            else if (pinState === 'right') rightPinnedCols.push(col);
            else normalCols.push(col);
        });
        const orderedColumns = [...leftPinnedCols, ...normalCols, ...rightPinnedCols];

        const headerHtml = orderedColumns.map(col => {
            const pinState = meterDetailTablePinningState[col.header];
            let pinContainerClass = '', pinIconClass = '';
            if (pinState === 'left') { pinContainerClass = 'pinned pinned-left'; pinIconClass = 'active-left'; }
            else if (pinState === 'right') { pinContainerClass = 'pinned pinned-right'; pinIconClass = 'active-right'; }
            const pinIcon = col.pinnable ? `<i class="fas fa-thumbtack pin-icon ${pinIconClass}" data-column-header="${col.header}" data-table-type="detail" onclick="showMeterPinningMenu(this, event)"></i>` : '';
            return `<th class="${col.class || ''} ${pinContainerClass}">${col.header}${pinIcon}</th>`;
        }).join('');
        container.parent().find('thead').html(`<tr>${headerHtml}</tr>`);

        // --- Body ---
        const bodyHtml = sortedRooms.map(room => {
            const currentReading = readingsMap.get(room.id) || {};
            const lastWater = findPreviousReading(monthYear, room.id, 'water');
            const lastElectric = findPreviousReading(monthYear, room.id, 'electric');

            const displayTenantName = room.tenantNameSnapshot;
            const tenantHtml = (displayTenantName && displayTenantName !== '') ? `<span class="text-dark">${displayTenantName}</span>` : '<em class="text-muted">ห้องว่าง</em>';
            const lastUpdatedHtml = currentReading.lastUpdated ? formatThaiDateTime(currentReading.lastUpdated) : '<span class="text-muted">-</span>';

            let roomNumberDisplay = room.number;
            // ✅ [เพิ่ม] แสดงสัญลักษณ์ถ้าเป็นห้องที่ถูกลบ
            if (room.isDeleted) roomNumberDisplay = `<span class="text-danger" title="ห้องนี้ถูกลบไปแล้ว แต่มีประวัติการจด">${room.number} <i class="fas fa-trash-alt small"></i></span>`;

            // ... [ส่วนคำนวณ Water/Electric เหมือนเดิม] ...
            let waterStartVal = parseInt(currentReading.waterbefore, 10);
            if (isNaN(waterStartVal) && lastWater) waterStartVal = lastWater.value;
            let waterAlertHtml = '';
            if (lastWater && lastWater.isSkipped && isNaN(parseInt(currentReading.waterbefore, 10))) {
                waterAlertHtml = `<i class="fas fa-history text-warning ms-1" style="font-size: 0.8em; cursor: help;" title="ยอดข้ามเดือน มาจาก ${lastWater.monthYear}"></i>`;
            }

            let electricStartVal = parseInt(currentReading.electricbefore, 10);
            if (isNaN(electricStartVal) && lastElectric) electricStartVal = lastElectric.value;
            let electricAlertHtml = '';
            if (lastElectric && lastElectric.isSkipped && isNaN(parseInt(currentReading.electricbefore, 10))) {
                electricAlertHtml = `<i class="fas fa-history text-warning ms-1" style="font-size: 0.8em; cursor: help;" title="ยอดข้ามเดือน มาจาก ${lastElectric.monthYear}"></i>`;
            }

            const waterCurrentUsage = calculateMeterUsage(waterStartVal, parseInt(currentReading.waterReading, 10));
            const electricCurrentUsage = calculateMeterUsage(electricStartVal, parseInt(currentReading.electricReading, 10));
            
            const prevWaterUsage = findPreviousUsageAmount(monthYear, room.id, 'water');
            const waterTrendHtml = getTrendHtml(waterCurrentUsage, prevWaterUsage);
            const prevElectricUsage = findPreviousUsageAmount(monthYear, room.id, 'electric');
            const electricTrendHtml = getTrendHtml(electricCurrentUsage, prevElectricUsage);

            const rowData = {
                roomNumber: roomNumberDisplay,
                tenant: tenantHtml,
                waterPrev: createMeterHtml(room.meterWaterActive, waterStartVal) + waterAlertHtml,
                waterCurr: createMeterHtml(room.meterWaterActive, currentReading.waterReading),
                waterUnit: createUsageHtml(room.meterWaterActive, waterCurrentUsage, waterTrendHtml),
                electricPrev: createMeterHtml(room.meterElectricActive, electricStartVal) + electricAlertHtml,
                electricCurr: createMeterHtml(room.meterElectricActive, currentReading.electricReading),
                electricUnit: createUsageHtml(room.meterElectricActive, electricCurrentUsage, electricTrendHtml),
                lastUpdated: lastUpdatedHtml
            };

            const cellsHtml = orderedColumns.map(col => {
                const pinState = meterDetailTablePinningState[col.header];
                let pinClass = '';
                if (pinState === 'left') pinClass = 'pinned pinned-left';
                else if (pinState === 'right') pinClass = 'pinned pinned-right';
                return `<td class="${col.class || ''} ${pinClass}">${rowData[col.key]}</td>`;
            }).join('');

            return `<tr data-room-id="${room.id}">${cellsHtml}</tr>`;
        }).join('');

        container.html(bodyHtml);
        recalculateMeterTableOffsets(container.parent());
    }

    function setupFormForMonth(monthYear) {
        meterPageTempCache = {};

        $('#meter-reading-form-container').empty();
        const isNew = !monthYear;
        const monthSelect = $('#meterMonthSelect');
        const yearSelect = $('#meterYearSelect');
        const title = document.getElementById('meter-form-title');

        if (isNew) {
            title.textContent = 'เพิ่มจดบันทึกใหม่';

            // เปิดการใช้งาน และ Trigger Select2
            yearSelect.prop('disabled', false).trigger('change.select2');
            monthSelect.prop('disabled', false).trigger('change.select2');

            // ตั้งค่าปีปัจจุบัน
            const today = new Date();
            yearSelect.val(today.getFullYear() + 543).trigger('change'); // Trigger change เพื่อให้ไปคำนวณเดือนใหม่

            // updateAvailableMonths() จะถูกเรียกผ่าน Event Listener ของ yearSelect ที่เราผูกไว้

            // เช็คว่าเดือนว่างไหม
            setTimeout(() => {
                if (monthSelect.prop('disabled')) {
                    renderMeterFormTable(null, null);
                    Swal.fire('แจ้งเตือน', 'ปีนี้มีการจดบันทึกครบทุกเดือนแล้ว', 'info');
                } else {
                    renderMeterFormTable(null, null);
                }
            }, 50);

        } else {
            // โหมดแก้ไข
            const [month, year] = monthYear.split('/');
            title.textContent = `แก้ไข: ${monthYearToText(monthYear)}`;

            // ตั้งค่าปี และ Trigger Select2
            yearSelect.val(parseInt(year) + 543).trigger('change.select2');

            // บังคับให้ Dropdown มีเดือนนี้อยู่
            updateAvailableMonths(month);

            // ตั้งค่าเดือน และ Trigger Select2
            monthSelect.val(month).trigger('change.select2');

            const group = allMeterReadings.find(g => g.monthYear === monthYear);

            // ปิดการใช้งาน (Disable)
            monthSelect.prop('disabled', true);
            yearSelect.prop('disabled', true);

            renderMeterFormTable(monthYear, group);
        }
    }


function renderMeterFormTable(monthYear, groupData) {
        const container = $('#meter-reading-form-container');

        // 1. SAVE STATE (คงเดิม)
        container.find('input[data-room-id]').each(function () {
            const rId = $(this).data('room-id');
            const type = $(this).data('type');
            const val = $(this).val();
            if (type === 'water' || type === 'electric') {
                if (!meterPageTempCache[rId]) meterPageTempCache[rId] = {};
                meterPageTempCache[rId][type] = val;
            }
        });

        // CLEANUP Select2 (คงเดิม)
        container.find('select.tenant-selector').each(function () {
            if ($(this).data('select2')) $(this).select2('destroy');
        });
        $('.meter-temp-dropdown').closest('.select2-container').remove();
        container.parent().addClass('meter-table');

        // Prepare Tenants Options (คงเดิม)
        const allTenants = window.tenantData || [];
        let tenantOptionsBase = '<option value="">-- ว่าง / ระบุใหม่ --</option>';
        allTenants.forEach(t => {
            tenantOptionsBase += `<option value="${t.id}">${t.name} (ID: ${t.id})</option>`;
        });

        // ✅ [แก้ไข] Prepare Rooms Logic
        let roomsToRender = [];
        
        if (groupData && groupData.data && groupData.data.length > 0) {
            // --- กรณีแก้ไข (Edit Mode): ยึดข้อมูลจาก Data เก่า (groupData) เป็นหลัก ---
            // เพื่อให้ห้องที่ถูกลบไปแล้ว แต่เคยมีการจดบันทึกไว้ ยังคงแสดงขึ้นมาให้แก้ไขได้
            roomsToRender = groupData.data.map(reading => {
                const realRoom = window.allRooms.find(r => r.id === reading.roomId);
                if (realRoom) {
                    // ห้องยังมีอยู่จริง: ใช้ข้อมูลล่าสุดจาก Room Setting ผสมกับข้อมูลที่บันทึกไว้
                    return {
                        ...realRoom,
                        savedTenantId: reading.customerId,
                        savedTenantName: reading.tenantName,
                        meterWaterActive: reading.waterActive !== undefined ? reading.waterActive : realRoom.meterWaterActive,
                        meterElectricActive: reading.electricActive !== undefined ? reading.electricActive : realRoom.meterElectricActive
                    };
                } else {
                    // ห้องถูกลบไปแล้ว: ใช้ข้อมูลจาก Data เก่า (reading) สร้าง object จำลองขึ้นมา
                    return {
                        id: reading.roomId,
                        number: reading.roomNumber || 'ห้องเก่า', // ดึงเลขห้องเก่ามาแสดง
                        towerId: 'deleted',
                        savedTenantId: reading.customerId,
                        savedTenantName: reading.tenantName,
                        meterWaterActive: reading.waterActive !== false,
                        meterElectricActive: reading.electricActive !== false,
                        isDeleted: true // Flag ไว้แสดง UI ว่าห้องถูกลบ
                    };
                }
            });
        } else {
            // --- กรณีจดใหม่ (New Mode): ดูจากห้องที่มีอยู่จริงเท่านั้น ---
            roomsToRender = window.allRooms.filter(r => r.isActive !== false).map(r => ({
                ...r,
                savedTenantId: r.tenant ? r.tenant.id : null,
                savedTenantName: r.tenant ? r.tenant.name : null
            }));
        }

        // --- กำหนดเดือนที่จะใช้ค้นหาย้อนหลัง (คงเดิม) ---
        let targetMonthYear = monthYear;
        if (!targetMonthYear) {
            const m = $('#meterMonthSelect').val();
            const y = parseInt($('#meterYearSelect').val()) - 543;
            targetMonthYear = `${m}/${y}`;
        }

        const readings = groupData?.data || [];
        const readingsMap = new Map(readings.map(r => [r.roomId, r]));

        // --- Filtering Logic (คงเดิม) ---
        const search = document.getElementById('searchMeterRoom').value.toLowerCase();
        const selectedTowers = $('#filterMeterTower').val();
        const filteredRooms = roomsToRender.filter(room => {
            if (selectedTowers && selectedTowers.length > 0) {
                const towerId = room.towerId || 'unassigned';
                if (room.isDeleted && !selectedTowers.includes('deleted')) {
                    // ถ้าเป็นห้องที่ลบ และไม่ได้ติ๊กเลือก deleted ให้ซ่อน
                } else if (!selectedTowers.includes(towerId)) return false;
            }
            if (search) {
                const tName = room.savedTenantName || '';
                if (!String(room.number).toLowerCase().includes(search) && !tName.toLowerCase().includes(search)) return false;
            }
            return true;
        });

        if (filteredRooms.length === 0) {
            container.html('<tr><td colspan="100%" class="text-center text-muted p-5">ไม่พบห้องพักที่ตรงตามเงื่อนไข</td></tr>');
            return;
        }

        const sortedFilteredRooms = filteredRooms.sort((a, b) => String(a.number).localeCompare(String(b.number), undefined, { numeric: true }));

        // ... [ส่วน Header Config และ Render เหมือนเดิม] ...
        const columnConfig = [
            { header: 'เลขห้อง', key: 'roomNumber', class: 'text-center fw-bold', pinnable: true },
            { header: 'ผู้เช่า', key: 'tenantSelect', pinnable: true },
            // --- [แก้ไข] สลับเอา ไฟ ขึ้นก่อน น้ำ ---
            { header: 'ไฟก่อน', key: 'electricPrev', class: 'text-end', pinnable: true },
            { header: 'ไฟนี้', key: 'electricCurr', class: 'text-end', pinnable: true },
            { header: 'หน่วยไฟ', key: 'electricUnit', class: 'text-end', pinnable: true },
            { header: 'น้ำก่อน', key: 'waterPrev', class: 'text-end', pinnable: true },
            { header: 'น้ำนี้', key: 'waterCurr', class: 'text-end', pinnable: true },
            { header: 'หน่วยน้ำ', key: 'waterUnit', class: 'text-end', pinnable: true },
            // ------------------------------------
            { header: 'จัดการ', key: 'actions', class: 'text-center', pinnable: true }
        ];

        // ... [ส่วนจัดเรียง Column Header คงเดิม] ...
        const leftPinnedCols = []; const rightPinnedCols = []; const normalCols = [];
        columnConfig.forEach(col => {
            const pinState = meterTablePinningState[col.header];
            if (pinState === 'left') leftPinnedCols.push(col);
            else if (pinState === 'right') rightPinnedCols.push(col);
            else normalCols.push(col);
        });
        const orderedColumns = [...leftPinnedCols, ...normalCols, ...rightPinnedCols];

        const headerHtml = orderedColumns.map(col => {
            const pinState = meterTablePinningState[col.header];
            let pinContainerClass = '', pinIconClass = '';

    function recalculateMeterTableOffsets($table) {
        // ใช้ setTimeout เพื่อให้มั่นใจว่า DOM render เสร็จแล้วจึงคำนวณความกว้าง
        setTimeout(() => {
            const $headers = $table.find('thead th');
            const $rows = $table.find('tbody tr');

            // คำนวณฝั่งซ้าย (Left Pinned)
            let leftOffset = 0;
            $headers.filter('.pinned-left').each(function () {
                const $th = $(this);
                $th.css('left', `${leftOffset}px`);

                const colIndex = $th.index();
                $rows.each(function () {
                    $(this).find('td').eq(colIndex).css('left', `${leftOffset}px`);
                });

                leftOffset += $th.outerWidth();
            });

            // คำนวณฝั่งขวา (Right Pinned)
            let rightOffset = 0;
            // ต้องย้อนกลับ (Reverse) เพราะ right: 0 เริ่มจากขวาสุด
            $($headers.filter('.pinned-right').get().reverse()).each(function () {
                const $th = $(this);
                $th.css('right', `${rightOffset}px`);

                const colIndex = $th.index();
                $rows.each(function () {
                    $(this).find('td').eq(colIndex).css('right', `${rightOffset}px`);
                });

                rightOffset += $th.outerWidth();
            });
        }, 0);
    }

    // ฟังก์ชันแสดงเมนูเมื่อคลิกไอคอน Pin
    function showMeterPinningMenu(iconElement, event) {
        event.stopPropagation();
        event.preventDefault();

        $('.pinning-menu').remove();

        const $icon = $(iconElement);
        const columnHeader = $icon.data('column-header');

        // 1. ตรวจสอบว่าเป็นตารางไหน (Detail หรือ Form)
        const tableType = $icon.data('table-type') || 'form'; // default เป็น form ถ้าไม่มีค่า

        // 2. เลือก State ที่ถูกต้อง
        let targetStateObj;
        if (tableType === 'detail') {
            targetStateObj = meterDetailTablePinningState;
        } else {
            targetStateObj = meterTablePinningState;
        }

        const currentState = targetStateObj[columnHeader];

        const menuHtml = `
  <ul class="pinning-menu">
   <li data-action="left" class="${currentState === 'left' ? 'disabled' : ''}"><i class="fas fa-chevron-left fa-fw"></i> ปักหมุดซ้าย</li>
   <li data-action="right" class="${currentState === 'right' ? 'disabled' : ''}"><i class="fas fa-chevron-right fa-fw"></i> ปักหมุดขวา</li>
   <hr class="my-1">
   <li data-action="unpin" class="${!currentState ? 'disabled' : ''}"><i class="fas fa-times fa-fw"></i> ปลดหมุด</li>
  </ul>
 `;

        const $menu = $(menuHtml);
        $('body').append($menu);

        const iconOffset = $icon.offset();
        $menu.css({
            top: iconOffset.top + $icon.outerHeight() + 5,
            left: iconOffset.left - $menu.outerWidth() + $icon.outerWidth()
        });

        $menu.find('li:not(.disabled)').on('click', function () {
            const action = $(this).data('action');

            // 3. อัปเดต State ให้ถูกตัวแปร
            targetStateObj[columnHeader] = (action === 'unpin') ? false : action;

            // 4. Render ตารางที่ถูกต้อง
            if (tableType === 'detail') {
                // Render Detail Table
                if (currentMeterState.view === 'details' && currentMeterState.monthYear) {
                    renderMeterDetailTable(currentMeterState.monthYear);
                }
            } else {
                // Render Form Table (Logic เดิม)
                const currentMonthYear = currentMeterState.monthYear;
                const currentGroup = allMeterReadings.find(g => g.monthYear === currentMonthYear);
                renderMeterFormTable(currentMonthYear, currentGroup);
            }

            $menu.remove();
        });

        setTimeout(() => {
            $(window).one('click.meterPinMenu', function () {
                $menu.remove();
            });
        }, 0);
    }

    function updateUsageCalculation(event) {
        const input = event.target;
        const roomId = input.dataset.roomId;
        const type = input.dataset.type;

        let currentValueStr = input.value.replace(/[^0-9]/g, '');
        if (currentValueStr.length > 4) currentValueStr = currentValueStr.slice(0, 4);
        if (input.value !== currentValueStr) input.value = currentValueStr;
        const currentValue = parseInt(currentValueStr, 10);

        const prevInput = document.querySelector(`input[data-room-id="${roomId}"][data-type="${type}-prev"]`);
        const usageDisplayContainer = document.querySelector(`span[data-room-id="${roomId}"][data-type="${type}-usage"]`);

        if (!prevInput || !usageDisplayContainer) return;

        const prevValue = parseInt(prevInput.value, 10);

        const usage = calculateMeterUsage(prevValue, currentValue);

        const priorUsageStr = usageDisplayContainer.dataset.priorUsage;
        const priorUsage = (priorUsageStr && priorUsageStr !== '') ? parseInt(priorUsageStr, 10) : null;

        if (usage !== null) {
            const trendHtml = getTrendHtml(usage, priorUsage);
            usageDisplayContainer.innerHTML = `${usage}${trendHtml}`;
            usageDisplayContainer.classList.add('text-primary', 'fw-bold');
        } else {
            usageDisplayContainer.innerHTML = '-';
            usageDisplayContainer.classList.remove('text-primary', 'fw-bold');
        }
    }

    function parseInputAsInt(value) {
        // ... (ฟังก์ชันนี้ไม่เปลี่ยนแปลง) ...
        if (value === null || value === undefined) return null;
        const strValue = String(value).trim();
        if (strValue === '') return null;
        const parsedValue = parseInt(strValue, 10);
        return isNaN(parsedValue) ? null : parsedValue;
    }

    // --- [3] เพิ่มฟังก์ชันใหม่ `formatThaiDateTime` ---
    function formatThaiDateTime(isoString) {
        if (!isoString) return null;
        try {
            const date = new Date(isoString);
            if (isNaN(date.getTime())) {
                return isoString; // ถ้าแปลงไม่ได้ คืนค่าเดิม
            }

            // options สำหรับ 'th-TH' (Asia/Bangkok)
            const options = {
                year: 'numeric',
                month: '2-digit',
                day: '2-digit',
                hour: '2-digit',
                minute: '2-digit',
                timeZone: 'Asia/Bangkok',
                hour12: false,
                calendar: 'buddhist' // ใช้ พ.ศ.
            };

            // .replace เพื่อจัดรูปแบบ "DD/MM/YYYY, HH:mm"
            return date.toLocaleString('th-TH', options).replace(',', ' ');

        } catch (e) {
            console.error("Error formatting date:", e);
            return isoString; // คืนค่าเดิมถ้ามีปัญหา
        }
    }

    function monthYearToText(monthYear) {
        if (!monthYear || typeof monthYear !== 'string') return "เดือน/ปี ไม่ถูกต้อง";

        let month, year;

        if (monthYear.includes('/')) {
            const parts = monthYear.split('/');
            month = parseInt(parts[0], 10);
            year = parseInt(parts[1], 10);
        } else if (monthYear.includes('-')) {
            const parts = monthYear.split('-');
            year = parseInt(parts[0], 10);
            month = parseInt(parts[1], 10);
        } else {
            return `รูปแบบผิด: ${monthYear}`;
        }

        if (isNaN(month) || isNaN(year)) {
            return `ข้อมูลผิด: ${monthYear}`;
        }

        const thaiMonths = [
            "มกราคม", "กุมภาพันธ์", "มีนาคม", "เมษายน", "พฤษภาคม", "มิถุนายน",
            "กรกฎาคม", "สิงหาคม", "กันยายน", "ตุลาคม", "พฤศจิกายน", "ธันวาคม"
        ];

        const thaiMonth = thaiMonths[month - 1];
        // ⭐️ แปลง ค.ศ. เป็น พ.ศ.
        const thaiYear = year + 543;

        if (!thaiMonth) {
            return `เดือนผิด: ${monthYear}`;
        }

        return `${thaiMonth} ${thaiYear}`;
    }

    function debounce(func, delay) {
        // ... (ฟังก์ชันนี้ไม่เปลี่ยนแปลง) ...
        let timeout;
        return function (...args) {
            clearTimeout(timeout);
            timeout = setTimeout(() => func.apply(this, args), delay);
        };
    }

function prepareAndSendImport(validRows) {
        // จัดกลุ่มตามเดือน
        const importDataByMonth = {};

        validRows.forEach(row => {
            const monthYear = row.monthYear;

            let targetCustomerId = null;
            if (row.csvTenantId) {
                targetCustomerId = row.csvTenantId;
            } else if (row.roomObj && row.roomObj.tenant) {
                targetCustomerId = row.roomObj.tenant.id;
            }

            const targetTenantName = row.tenantName || (row.roomObj && row.roomObj.tenant ? row.roomObj.tenant.name : '');

            const readingObj = {
                roomId: row.roomId,
                roomNumber: row.roomNumber, // ✅ [เพิ่ม] บันทึก Snapshot เลขห้อง
                towerId: row.towerId || 'unassigned', // ✅ [เพิ่ม] บันทึก Snapshot ตึก
                waterReading: row.waterVal,
                electricReading: row.electricVal,
                waterActive: true,
                electricActive: true,
                customerId: targetCustomerId,
                tenantName: targetTenantName,
                lastUpdated: new Date().toISOString()
            };

            if (!importDataByMonth[monthYear]) {
                importDataByMonth[monthYear] = [];
            }
            importDataByMonth[monthYear].push(readingObj);
        });

        // ปิด Modal
        const modalEl = document.getElementById('meterImportModal');
        const modalInstance = bootstrap.Modal.getInstance(modalEl);
        if (modalInstance) modalInstance.hide();

        // เรียกฟังก์ชันส่ง Server เดิม
        performBatchImport(importDataByMonth);
    }

            if (tenantInfo.includes(',')) {

    function parseFlexibleMonthYear(str) {
        if (!str) return null;

        // 1. ล้างค่าขยะ และเปลี่ยนชื่อเดือนภาษาอังกฤษเป็นตัวเลข
        let s = str.replace(/^\uFEFF/, '').trim().toLowerCase();
        const months = {
            'jan': '01', 'feb': '02', 'mar': '03', 'apr': '04', 'may': '05', 'jun': '06',
            'jul': '07', 'aug': '08', 'sep': '09', 'oct': '10', 'nov': '11', 'dec': '12'
        };
        Object.keys(months).forEach(key => {
            if (s.includes(key)) s = s.replace(key, months[key]);
        });

        // 2. เปลี่ยนตัวคั่นเป็น /
        s = s.replace(/[-.]/g, '/');
        const parts = s.split('/');

        let m, y;

        // --- กรณีที่ 1: มี 3 ส่วน (เช่น 1/1/2025 หรือ 2025-01-01) ---
        if (parts.length === 3) {
            const p0 = parseInt(parts[0], 10);
            const p1 = parseInt(parts[1], 10);
            const p2 = parseInt(parts[2], 10);

            if (p0 > 1000) {
                // รูปแบบ ปี-เดือน-วัน (2025-01-01)
                y = p0; m = p1;
            } else if (p2 > 1000) {
                // รูปแบบ วัน/เดือน/ปี (1/1/2025) หรือ เดือน/วัน/ปี
                y = p2;
                // โดยทั่วไป Excel ไทยจะส่งมาเป็น วัน/เดือน/ปี -> เอาค่ากลางเป็นเดือน
                m = p1;
            } else if (p2 > 2400) {
                // กรณีปี พ.ศ. (1/1/2568)
                y = p2 - 543;
                m = p1;
            }
        }
        // --- กรณีที่ 2: มี 2 ส่วน (เช่น 01/2025, Jan-25) ---
        else if (parts.length === 2) {
            m = parseInt(parts[0], 10);
            y = parseInt(parts[1], 10);

            // กรณีสลับที่ (2025/01)
            if (m > 100) { const temp = m; m = y; y = temp; }
        }
        else {
            return null;
        }

        // 3. ปรับปี 2 หลัก (เช่น 25 -> 2025)
        if (y < 100) y += 2000;

        // 4. รองรับปี พ.ศ. (เผื่อหลุดมา) -> แปลงเป็น ค.ศ.
        if (y > 2400) y -= 543;

        // ตรวจสอบความถูกต้อง
        if (!isNaN(m) && m >= 1 && m <= 12 && !isNaN(y) && y > 2000) {
            return `${String(m).padStart(2, '0')}/${y}`;
        }

        return null;
    }

function processCSVImport(csvText) {
        const lines = csvText.split(/\r\n|\n|\r/);

        if (lines.length < 2) {
            Swal.fire('ไฟล์ผิดพลาด', 'ไฟล์ CSV ไม่มีข้อมูล', 'error');
            return;
        }

        const existingMonths = new Set(allMeterReadings.map(g => g.monthYear));

        currentImportRawData = [];

        // เริ่มอ่านตั้งแต่บรรทัดที่ 2 (index 1)
        for (let i = 1; i < lines.length; i++) {
            const line = lines[i].trim();
            if (!line) continue;

            const parts = line.split(',').map(p => p.replace(/"/g, '').trim());
            if (parts.length < 5) continue;

            const rawMonthYear = parts[0];
            const roomNumber = parts[1];

            // --- ส่วนรับค่า Tenant ---
            const rawTenantInput = parts[2];
            let finalTenantName = rawTenantInput;
            let finalTenantId = null;

            if (rawTenantInput && window.tenantData && Array.isArray(window.tenantData)) {
                const foundTenant = window.tenantData.find(t => String(t.id) === rawTenantInput);
                if (foundTenant) {
                    finalTenantName = foundTenant.name;
                    finalTenantId = foundTenant.id;
                }
            }

            // ✅ [แก้ไข] สลับตำแหน่งกลับให้ถูกต้อง (น้ำมาก่อน หรือตามไฟล์ CSV ของคุณ)
            // ถ้าไฟล์ CSV ของคุณเรียงเป็น: เดือน, ห้อง, ผู้เช่า, น้ำ, ไฟ
            const waterValStr = parts[3];    // ช่องที่ 4 คือ น้ำ
            const electricValStr = parts[4]; // ช่องที่ 5 คือ ไฟ

            const validFormatDate = parseFlexibleMonthYear(rawMonthYear);
            const room = window.allRooms.find(r => r.number === roomNumber);

            let status = { isValid: true, message: 'ปกติ' };

            // --- เริ่มตรวจสอบเงื่อนไข ---
            if (!room) {
                status.isValid = false; 
                status.message = 'ไม่พบเลขห้องนี้ในระบบปัจจุบัน';
            }
            else if (!validFormatDate) {
                status.isValid = false;
                status.message = `รูปแบบเดือน/ปี ไม่ถูกต้อง (${rawMonthYear})`;
            }
            else if (existingMonths.has(validFormatDate)) {
                status.isValid = false;
                status.message = `มีข้อมูลเดือน ${validFormatDate} ในระบบแล้ว`;
            }
            else {
                const waterValCheck = waterValStr !== '' ? parseInt(waterValStr) : null;
                const electricValCheck = electricValStr !== '' ? parseInt(electricValStr) : null;

                if (waterValStr !== '' && isNaN(waterValCheck)) {
                    status.isValid = false;
                    status.message = 'ค่าน้ำต้องเป็นตัวเลข';
                }
                else if (electricValStr !== '' && isNaN(electricValCheck)) {
                    status.isValid = false;
                    status.message = 'ค่าไฟต้องเป็นตัวเลข';
                }
            }

            const waterVal = waterValStr !== '' ? parseInt(waterValStr) : null;
            const electricVal = electricValStr !== '' ? parseInt(electricValStr) : null;

            currentImportRawData.push({
                monthYear: validFormatDate || rawMonthYear,
                roomNumber: roomNumber,
                tenantName: finalTenantName,
                csvTenantId: finalTenantId,
                waterVal,
                electricVal,
                roomId: room ? room.id : null,
                towerId: room ? room.towerId : null,
                roomObj: room,
                status: status
            });
        }

        if (currentImportRawData.length === 0) {
            Swal.fire('ไม่พบข้อมูล', 'ไม่สามารถอ่านข้อมูลจากไฟล์ได้', 'warning');
            return;
        }

        renderPreviewTable(currentImportRawData);

        document.getElementById('meter-import-view-upload').style.display = 'none';
        document.getElementById('meter-import-view-preview').style.display = 'block';
    }

    function getTrendHtml(currentUsage, priorUsage) {
        if (priorUsage === null || priorUsage === undefined || isNaN(priorUsage)) return '';

        const diff = currentUsage - priorUsage;

             <i class="fas fa-arrow-up small"></i> ${diff}

        } else if (diff < 0) {
            // ใช้น้อยลง (สีเขียว)
            return `<span class="text-success fw-bold" style="${style}" title="ใช้ลดลง ${Math.abs(diff)} หน่วย จากเดือนก่อน (${priorUsage})">
             <i class="fas fa-arrow-down small"></i> ${Math.abs(diff)}

        } else {
            // เท่าเดิม (สีเทา)
            return `<span class="text-muted" style="${style}" title="ใช้เท่าเดือนก่อน">
             <i class="fas fa-minus small"></i>
            </span>`;
        }
    }

    function calculateMeterUsage(prev, current) {
        if (prev === null || prev === undefined || isNaN(prev)) return null;
        if (current === null || current === undefined || isNaN(current)) return null;

        let p = parseInt(prev, 10);
        let c = parseInt(current, 10);

        if (c >= p) {
            return c - p; // เคสปกติ
        } else {
            return (c + 10000) - p; // เคสวนรอบ (Roll over)
        }
    }

    function getMonthYearValue(monthYear) {
        if (!monthYear) return 0;
        try {
            let parts, m, y;
            if (monthYear.includes('/')) parts = monthYear.split('/');
            else if (monthYear.includes('-')) parts = monthYear.split('-');
            else return 0;

            if (parts[0].length === 4) { // YYYY-MM
                y = parseInt(parts[0]); m = parseInt(parts[1]);
            } else { // MM/YYYY
                m = parseInt(parts[0]); y = parseInt(parts[1]);
            }

            // แปลง พ.ศ. เป็น ค.ศ. ถ้าจำเป็น (สันนิษฐานว่าถ้า > 2400 คือ พ.ศ.)
            const realYear = y > 2400 ? y - 543 : y;
            return (realYear * 100) + m;
        } catch (e) {
            return 0;
        }
    }

    function findPreviousReading(currentMonthYear, roomId, type) {
        const currentVal = getMonthYearValue(currentMonthYear);
        if (currentVal === 0) return null;

        // กรองเอาเฉพาะเดือนที่เก่ากว่าเดือนปัจจุบัน และเรียงจาก ใหม่ -> เก่า
        const previousGroups = allMeterReadings
            .filter(g => getMonthYearValue(g.monthYear) < currentVal)
            .sort((a, b) => getMonthYearValue(b.monthYear) - getMonthYearValue(a.monthYear));

        for (let group of previousGroups) {
            const reading = (group.data || []).find(r => r.roomId === roomId);
            if (reading) {
                let val = null;
                // เช็คค่าตามประเภท
                if (type === 'water' && reading.waterReading !== null && reading.waterReading !== "" && reading.waterReading !== undefined) {
                    val = parseInt(reading.waterReading, 10);
                } else if (type === 'electric' && reading.electricReading !== null && reading.electricReading !== "" && reading.electricReading !== undefined) {
                    val = parseInt(reading.electricReading, 10);
                }

                if (!isNaN(val) && val !== null) {
                    // คำนวณความห่างของเดือน
                    const prevVal = getMonthYearValue(group.monthYear);

                    // logic ง่ายๆ: เช็คว่าห่างกันเกิน 1 step หรือไม่ (เช่น 202503 - 202501 = 2 -> skipped)
                    // (วิธีนี้ใช้ได้กรณีปีเดียวกัน ถ้าข้ามปีต้องใช้ Date Object แต่เพื่อความเร็วใช้ diff value ไปก่อน)
                    // เพื่อความชัวร์ ใช้ Date Object คำนวณ Gap
                    const cY = Math.floor(currentVal / 100); const cM = currentVal % 100;
                    const pY = Math.floor(prevVal / 100); const pM = prevVal % 100;
                    const totalMonthsDiff = (cY - pY) * 12 + (cM - pM);

                    return {
                        value: val,
                        monthYear: group.monthYear,
                        isSkipped: totalMonthsDiff > 1, // ถ้าห่าง > 1 เดือน แสดงว่ามีการกระโดดข้าม
                        gap: totalMonthsDiff
                    };
                }
            }
        }
        return null; // ไม่เจอเลย (อาจเป็นเดือนแรกสุด)
    }

    function findPreviousUsageAmount(currentMonthYear, roomId, type) {
        const currentVal = getMonthYearValue(currentMonthYear);
        if (currentVal === 0) return null;

        // 1. ดึงประวัติทั้งหมดที่เก่ากว่าเดือนปัจจุบัน เรียงจาก ใหม่ -> เก่า
        const previousGroups = allMeterReadings
            .filter(g => getMonthYearValue(g.monthYear) < currentVal)
            .sort((a, b) => getMonthYearValue(b.monthYear) - getMonthYearValue(a.monthYear));

        if (previousGroups.length === 0) return null; // ไม่มีประวัติเลย

        // 2. ดูข้อมูลของ "เดือนที่แล้ว" (Month - 1)
        const prevGroup = previousGroups[0];
        const prevReading = (prevGroup.data || []).find(r => r.roomId === roomId);

        if (!prevReading) return null;

        // หาค่าจดปลายเดือน (End Value) ของเดือนที่แล้ว
        let endVal, active;
        if (type === 'water') {
            endVal = prevReading.waterReading;
            active = prevReading.waterActive;
        } else {
            endVal = prevReading.electricReading;
            active = prevReading.electricActive;
        }

        // ถ้าเดือนที่แล้วปิดใช้งาน หรือไม่มีเลขจด ก็จบคำนวณไม่ได้
        if (active === false || endVal === null || endVal === "" || endVal === undefined) return null;
        endVal = parseInt(endVal, 10);

        // 3. หาค่าตั้งต้น (Start Value) ของเดือนที่แล้ว
        // วิธี A: ดูว่าใน DB บันทึก waterbefore/electricbefore ไว้ไหม (สำหรับข้อมูลใหม่ๆ)
        let startVal = type === 'water' ? prevReading.waterbefore : prevReading.electricbefore;

        // วิธี B: ถ้าไม่มี before (เช่น ข้อมูลเก่า) ให้วิ่งไปหาเลขจดของ "เดือนก่อนหน้านั้น" (Month - 2)
        if (startVal === null || startVal === undefined || isNaN(parseInt(startVal, 10))) {
            // วนลูปหากลุ่มถัดไปในลิสต์ (เริ่มที่ index 1 คือเดือนถัดไปจาก prevGroup)
            for (let i = 1; i < previousGroups.length; i++) {
                const olderGroup = previousGroups[i];
                const olderReading = (olderGroup.data || []).find(r => r.roomId === roomId);
                if (olderReading) {
                    let olderVal = type === 'water' ? olderReading.waterReading : olderReading.electricReading;
                    if (olderVal !== null && olderVal !== "" && olderVal !== undefined) {
                        startVal = parseInt(olderVal, 10);
                        break; // เจอแล้ว หยุดลูป
                    }
                }
            }
        } else {
            startVal = parseInt(startVal, 10);
        }

        // 4. คำนวณหน่วยที่ใช้ (ถ้ามีทั้ง ต้น และ ปลาย)
        if (!isNaN(startVal) && !isNaN(endVal)) {
            return calculateMeterUsage(startVal, endVal);
        }

        return null;
    }

function getMeterReadings() {
 try {
  const sheet = getSheet().getSheetByName(CONFIG.SHEETS.METER_READINGS);
  if (!sheet || sheet.getLastRow() < 2) {
   Logger.log("Meter readings sheet is empty or only has header.");
   return []; // ส่ง Array ว่างกลับไป
  }

  const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn()).getValues();
  Logger.log(`Found ${data.length} raw rows from sheet.`);

  const mappedData = data.map(row => {
   let readingData = [];
   try {
    if (row[CONFIG.METER_READINGS_SHEET.DATA]) {
     readingData = JSON.parse(row[CONFIG.METER_READINGS_SHEET.DATA]);
    }
   } catch (e) {
    Logger.log(`Could not parse meter data for row ${row[CONFIG.METER_READINGS_SHEET.ID]}: ${e}`);
   }
   
   // --- 1. แปลง monthYear (จากครั้งที่แล้ว) ---
   let monthYearValue = row[CONFIG.METER_READINGS_SHEET.MONTH_YEAR];
   if (monthYearValue instanceof Date) {
    const month = String(monthYearValue.getMonth() + 1).padStart(2, '0');
    const year = monthYearValue.getFullYear();
    monthYearValue = `${month}/${year}`;
   }

   // --- 2. แปลง recordDate (มีอยู่แล้ว) ---
   let recordDateValue = row[CONFIG.METER_READINGS_SHEET.RECORD_DATE];
   if (recordDateValue instanceof Date) {
    recordDateValue = recordDateValue.toISOString().split('T')[0];
   }
      
      // --- 3. แปลง createdAt (จุดแก้ไขสำคัญ) ---
      let createdAtValue = row[CONFIG.METER_READINGS_SHEET.CREATED_AT];
      if (createdAtValue instanceof Date) {
        createdAtValue = createdAtValue.toISOString(); // แปลงเป็น String
      }

      // --- 4. แปลง updatedAt (จุดแก้ไขสำคัญ) ---
      let updatedAtValue = row[CONFIG.METER_READINGS_SHEET.UPDATED_AT];
      if (updatedAtValue instanceof Date) {
        updatedAtValue = updatedAtValue.toISOString(); // แปลงเป็น String
      }

   return {
    id: row[CONFIG.METER_READINGS_SHEET.ID],
    monthYear: monthYearValue, // String "MM/YYYY" or ""
    data: readingData,
    recordDate: recordDateValue, // String "YYYY-MM-DD" or ""
    createdAt: createdAtValue, // String (ISO)
    updatedAt: updatedAtValue  // String (ISO)
   };
  });
    
  Logger.log(`Mapped data (before filter): ${JSON.stringify(mappedData.map(d => ({id: d.id, monthYear: d.monthYear, createdAt: d.createdAt})))}`); 

  const filteredData = mappedData.filter(group => group.id && group.monthYear);

  // Log ข้อมูลสุดท้ายที่จะส่งกลับไป
  Logger.log(`FINAL data being sent to client: ${JSON.stringify(filteredData)}`); 

  return filteredData;

 } catch (e) {
  // ดักจับข้อผิดพลาดทั้งหมดที่เกิดขึ้นในฟังก์ชัน
  Logger.log(`CRITICAL ERROR in getMeterReadings: ${e.message} \nStack: ${e.stack}`);
  // ส่ง Error กลับไปหา Client เพื่อให้ .withFailureHandler ทำงาน
  throw new Error(`ไม่สามารถดึงข้อมูลการจดมิเตอร์ได้: ${e.message}`);
 }
}

function saveMeterReading(monthYear, recordDate, dataArray) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(30000)) {
    throw new Error("ระบบไม่ว่าง โปรดลองอีกครั้ง (Server is busy)");
  }

  try {
    const sheet = getSheet().getSheetByName(CONFIG.SHEETS.METER_READINGS);
    const data = sheet.getDataRange().getValues();
    const now = new Date();
    
    // ค้นหาแถวที่มีอยู่สำหรับ monthYear นี้
    let rowIndex = -1;
    for (let i = 1; i < data.length; i++) {
      // ตรวจสอบค่า monthYear (อาจเป็น Date object หรือ String)
      let cellValue = data[i][CONFIG.METER_READINGS_SHEET.MONTH_YEAR];
      if (cellValue instanceof Date) {
         const month = String(cellValue.getMonth() + 1).padStart(2, '0');
         const year = cellValue.getFullYear();
         cellValue = `${month}/${year}`;
      }
      
      if (cellValue === monthYear) {
        rowIndex = i + 1; // 1-based index
        break;
      }
    }

    const dataJson = JSON.stringify(dataArray);

    if (rowIndex > -1) {
      // --- อัปเดตแถวที่มีอยู่ ---
      const rowId = sheet.getRange(rowIndex, CONFIG.METER_READINGS_SHEET.ID + 1).getValue();
      const createdAtValue = sheet.getRange(rowIndex, CONFIG.METER_READINGS_SHEET.CREATED_AT + 1).getValue();

      sheet.getRange(rowIndex, CONFIG.METER_READINGS_SHEET.DATA + 1).setValue(dataJson);
      sheet.getRange(rowIndex, CONFIG.METER_READINGS_SHEET.RECORD_DATE + 1).setValue(recordDate);
      sheet.getRange(rowIndex, CONFIG.METER_READINGS_SHEET.UPDATED_AT + 1).setValue(now);
      
      // [✅ แก้ไขจุดนี้] แปลง Date Object เป็น String ก่อนส่งกลับ
      return {
        id: rowId,
        monthYear: monthYear,
        data: dataArray,
        recordDate: recordDate,
        createdAt: (createdAtValue instanceof Date) ? createdAtValue.toISOString() : createdAtValue, // แปลง createdAt
        updatedAt: now.toISOString() // แปลง updatedAt
      };

    } else {
      // --- สร้างแถวใหม่ ---
      const lastId = sheet.getLastRow() > 1 ? sheet.getRange(sheet.getLastRow(), CONFIG.METER_READINGS_SHEET.ID + 1).getValue() : 0;
      const newId = String(parseInt(lastId) + 1);

      const newRow = new Array(sheet.getLastColumn()).fill('');
      newRow[CONFIG.METER_READINGS_SHEET.ID] = newId;
      newRow[CONFIG.METER_READINGS_SHEET.MONTH_YEAR] = monthYear;
      newRow[CONFIG.METER_READINGS_SHEET.DATA] = dataJson;
      newRow[CONFIG.METER_READINGS_SHEET.RECORD_DATE] = recordDate;
      newRow[CONFIG.METER_READINGS_SHEET.CREATED_AT] = now;
      newRow[CONFIG.METER_READINGS_SHEET.UPDATED_AT] = now;

      sheet.appendRow(newRow);
      sheet.getRange(sheet.getLastRow(), CONFIG.METER_READINGS_SHEET.ID + 1).setNumberFormat("@");
      
      // [✅ แก้ไขจุดนี้] แปลง Date Object เป็น String ก่อนส่งกลับ
      return {
        id: newId,
        monthYear: monthYear,
        data: dataArray,
        recordDate: recordDate,
        createdAt: now.toISOString(), // แปลง createdAt
        updatedAt: now.toISOString()  // แปลง updatedAt
      };
    }

  } catch (e) {
    Logger.log(`Error in saveMeterReading: ${e.message}`);
    throw new Error(`เกิดข้อผิดพลาดในการบันทึกข้อมูล: ${e.message}`);
  } finally {
    lock.releaseLock();
  }
  
}

function batchImportMeterReadings(payload) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(30000)) throw new Error("Server busy");

  try {
    const sheet = getSheet().getSheetByName(CONFIG.SHEETS.METER_READINGS);
    const data = sheet.getDataRange().getValues();
    const now = new Date();
    const recordDate = now.toISOString().split('T')[0];

    // สร้าง Map ของแถวที่มีอยู่ key = monthYear
    const existingRowMap = new Map();
    for (let i = 1; i < data.length; i++) {
      let mYear = data[i][CONFIG.METER_READINGS_SHEET.MONTH_YEAR];
      if (mYear instanceof Date) {
         mYear = Utilities.formatDate(mYear, Session.getScriptTimeZone(), "MM/yyyy");
      }
      existingRowMap.set(mYear, i + 1); // เก็บเลขแถว (1-based)
    }

    payload.forEach(item => {
      const monthYear = item.monthYear;
      const importData = item.data; // Array of readings
      const jsonStr = JSON.stringify(importData);

      if (existingRowMap.has(monthYear)) {
        // Update แถวเดิม
        const rowIndex = existingRowMap.get(monthYear);
        sheet.getRange(rowIndex, CONFIG.METER_READINGS_SHEET.DATA + 1).setValue(jsonStr);
        sheet.getRange(rowIndex, CONFIG.METER_READINGS_SHEET.UPDATED_AT + 1).setValue(now);
      } else {
        // Insert แถวใหม่
        const lastId = sheet.getLastRow() > 1 ? sheet.getRange(sheet.getLastRow(), CONFIG.METER_READINGS_SHEET.ID + 1).getValue() : 0;
        const newId = String(parseInt(lastId || 0) + 1);
        
        const newRow = new Array(sheet.getLastColumn()).fill('');
        newRow[CONFIG.METER_READINGS_SHEET.ID] = newId;
        newRow[CONFIG.METER_READINGS_SHEET.MONTH_YEAR] = monthYear;
        newRow[CONFIG.METER_READINGS_SHEET.DATA] = jsonStr;
        newRow[CONFIG.METER_READINGS_SHEET.RECORD_DATE] = recordDate;
        newRow[CONFIG.METER_READINGS_SHEET.CREATED_AT] = now;
        newRow[CONFIG.METER_READINGS_SHEET.UPDATED_AT] = now;
        
        sheet.appendRow(newRow);
        // อัปเดต Map เผื่อมีเดือนซ้ำใน Payload เดียวกัน (กันพลาด)
        existingRowMap.set(monthYear, sheet.getLastRow());
      }
    });

    return { success: true };

  } catch (e) {
    throw new Error("Batch Import Error: " + e.message);
  } finally {
    lock.releaseLock();
  }
}

/**
 * บันทึกรายการสูตรทั้งหมดลง Sheet "AllSetting"
 */
function saveSystemFormulas(formulasArray) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(10000)) throw new Error("Server busy");
  
  try {
    const sheet = getSheet().getSheetByName(CONFIG.SHEETS.ALL_SETTINGS);
    if (!sheet) throw new Error("Sheet 'AllSetting' not found");
    
    // 1. อ่านข้อมูล Config เดิมเก็บไว้ก่อน (ถ้ามี)
    let configRowData = null;
    const data = getData(sheet);
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][0]) === 'GLOBAL_CONFIG') {
        configRowData = data[i];
        break;
      }
    }

    // 2. ล้างข้อมูลเก่าทั้งหมด (ยกเว้น Header)
    const lastRow = sheet.getLastRow();
    if (lastRow > 1) {
      sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).clearContent();
    }
    
    const rowsToWrite = [];

    // 3. ใส่ Config เดิมกลับเข้าไปเป็นแถวแรก (ถ้ามี)
    if (configRowData) {
      // อัปเดตวันที่ล่าสุดหน่อยก็ได้ หรือใช้ค่าเดิม
      rowsToWrite.push(configRowData);
    }

    // 4. ใส่ข้อมูล Formula ใหม่ต่อท้าย
    formulasArray.forEach(f => {
      rowsToWrite.push([
        f.id,
        f.name,
        f.type,
        f.formula,
        f.isDefault,
        new Date()
      ]);
    });
    
    // 5. บันทึกลง Sheet ทีเดียว
    if (rowsToWrite.length > 0) {
      sheet.getRange(2, 1, rowsToWrite.length, rowsToWrite[0].length).setValues(rowsToWrite);
    }

    return true;
    
  } catch(e) {
    throw new Error("Error saving formulas: " + e.message);
  } finally {
    lock.releaseLock();
  }
}

// ฟังก์ชันแปลงค่า 16-bit register 2 ตัวให้เป็น Float (32-bit)
float getFloatFromRegisters(uint16_t reg1, uint16_t reg2) {
 union {
  uint32_t i;
  float f;
 } u;

 // การเรียงข้อมูล (Endianness)
 u.i = ((uint32_t)reg1 << 16) | reg2;
 return u.f;
}

// ฟังก์ชันอ่านค่า Analog จากโมดูล NTC เป็นองศาเซลเซียส
float getTemperatureNTC() {
  int analogValue = analogRead(NTC_PIN);
  Serial.print("Raw Analog Value: "); 
  Serial.println(analogValue); // ดูว่าค่าที่ได้คืออะไร (0-4095)

  if(analogValue >= 4090) {
    Serial.println("Error: Sensor Disconnected or Open Circuit!");
    return -999.0;
  }

 // คำนวณค่าความต้านทาน (อ้างอิง Voltage Divider 10k)
 float resistance = 10000.0 * (4095.0 / (float)analogValue - 1.0);

 // สมการ Steinhart-Hart คำนวณอุณหภูมิ (B-Value ~ 3950)
 float tempK = 1.0 / (1.0 / 298.15 + (1.0 / 3950.0) * log(resistance / 10000.0));
 return tempK - 273.15; // แปลงเคลวินเป็นเซลเซียส
}

// ฟังก์ชันย่อยสำหรับอ่านค่าและแสดงผล (เปลี่ยนให้ return ค่า float กลับมาด้วย)
float readMeterValue(uint16_t address, const char* label, const char* unit) {
 uint8_t result = node.readHoldingRegisters(address, 2);
 float value = -999.0; // ค่าเริ่มต้นกรณี Error

 if (result == node.ku8MBSuccess) {
  uint16_t reg1 = node.getResponseBuffer(0);
  uint16_t reg2 = node.getResponseBuffer(1);
  value = getFloatFromRegisters(reg1, reg2);
 
  Serial.print(label);
  Serial.print(": ");
  Serial.print(value);
  Serial.print(" ");
  Serial.println(unit);
 } else {
  Serial.print("Failed to read ");
  Serial.print(label);
  Serial.print(" (Error: 0x");
  Serial.print(result, HEX);
  Serial.println(")");
 }

 // หน่วงเวลาเล็กน้อยระหว่างการส่งคำสั่งแต่ละชุด เพื่อไม่ให้มิเตอร์ประมวลผลไม่ทัน
 delay(100);
 return value;
}

void setup() {
 Serial.begin(115200);

 // ตั้งค่าพิน MAX485
 pinMode(MAX485_RE_DE, OUTPUT);
 digitalWrite(MAX485_RE_DE, 0);

 // ตั้งค่าพิน Buzzer
 pinMode(BUZZER_PIN, OUTPUT);
 digitalWrite(BUZZER_PIN, HIGH); // สั่ง HIGH เพื่อให้ Buzzer เงียบก่อน

 // ตั้งค่า Baudrate
 Serial2.begin(9600, SERIAL_8N1, RX_PIN, TX_PIN);

 // ตั้งค่า Slave ID เป็น 1
 node.begin(1, Serial2);
 node.preTransmission(preTransmission);
 node.postTransmission(postTransmission);

 Serial.println("System Ready! Monitoring Meter and Temperature...");
 delay(2000);
}

void loop() {
 Serial.println("=============================");

 // --- 1. ส่วนอ่านค่าจากเซ็นเซอร์อุณหภูมิ NTC ---
 float tempC = getTemperatureNTC();
 if (tempC == -999.0) {
  Serial.println("Temp Sensor : Error / Disconnected!");
 } else {
  Serial.print("Cabinet Temp: ");
  Serial.print(tempC);
  Serial.println(" C");
 }

 // --- 2. ส่วนอ่านค่าจากมิเตอร์ไฟฟ้า (Modbus) ---
 // อ่านค่าหน่วยก้านไฟฟ้า (kWh)
 readMeterValue(0, "Total Energy  ", "kWh");

 Serial.println("--- Voltage ---");
 readMeterValue(100, "Phase A Voltage ", "V");
 readMeterValue(102, "Phase B Voltage ", "V");
 readMeterValue(104, "Phase C Voltage ", "V");

 Serial.println("--- Current ---");
 readMeterValue(106, "Phase A Current ", "A");
 readMeterValue(108, "Phase B Current ", "A");
 readMeterValue(110, "Phase C Current ", "A");

 Serial.println("--- Power & Freq ---");
 // เก็บค่ากำลังไฟฟ้ารวมไว้ในตัวแปร totalPower เพื่อไปเช็คเงื่อนไข
 float totalPower = readMeterValue(118, "Total Power  ", "kW");
 readMeterValue(144, "Frequency   ", "Hz");

 Serial.println("=============================\n");

 // --- 3. ส่วนระบบแจ้งเตือน (Alarm Logic) ---
 bool isOverheat = (tempC > MAX_TEMP && tempC != -999.0);
 bool isOverload = (totalPower > MAX_POWER && totalPower != -999.0);

 if (isOverheat || isOverload) {
  digitalWrite(BUZZER_PIN, LOW); // สั่ง LOW เพื่อให้ Buzzer ส่งเสียงดัง
 
  Serial.println("!!! ALARM TRIGGERED !!!");
  if (isOverheat) Serial.println("-> OVERHEAT DETECTED");
  if (isOverload) Serial.println("-> OVERLOAD DETECTED");
 } else {
  digitalWrite(BUZZER_PIN, HIGH); // สั่ง HIGH ให้ Buzzer เงียบ
 }

 // หน่วงเวลา 2 วินาทีก่อนเริ่มอ่านลูปใหม่ทั้งหมด
 delay(2000);
}

async function linkRichMenu(userId, richMenuId) {
    try {
        await axios.post(`https://api.line.me/v2/bot/user/${userId}/richmenu/${richMenuId}`, {}, {
            headers: { 'Authorization': `Bearer ${process.env.LINE_CHANNEL_ACCESS_TOKEN}` }
        });
        console.log(`✅ Switched menu for ${userId} to ${richMenuId}`);
    } catch (error) {
        console.error('❌ Error switching rich menu:', error.response?.data || error.message);
    }
}
